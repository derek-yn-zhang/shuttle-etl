[docs]def banner(text, width=60, char="="):
    """
    Print a centered text banner.

    Parameters
    ----------
    text: str
        Banner text.
    width: int
        Banner width.
    char: str
        Banner character.
    """

    print(f"{char*width}\n{f' {text} '.center(width, ' ')}\n{char*width}")
