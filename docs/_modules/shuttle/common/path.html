import os
from datetime import datetime



[docs]def stage(dataset, date: datetime):
    """
    Get the s3 staging address for a dataset on a given date.

    Parameters
    ----------
    dataset: str
        Extract name.
    date: datetime
        Extract date.

    Returns
    -------
    str
        s3 staging address.
    """

    return "{bucket_uri}/{prefix}/{dataset}_{date}/".format(
        bucket_uri=os.environ["bucket_uri"],
        prefix=os.environ["stage_prefix"],
        dataset=dataset,
        date=date.strftime("%Y_%m_%d"),
    )
