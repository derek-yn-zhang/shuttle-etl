    def __init__(self, config_file: str):
        self._job = None
        self.display = None
        self.config_file = config_file
        self.instance = self.get_instance_profile()
        self.config = self.parse_config(config_file)
        self.load_secrets(self.config, self.display)
        self.spark = self.get_spark_session()
        self.export_constants()
        self.print_config()

[docs]    def parse_config(self, config_file: str) -> DotMap:
        """
        Parse configuration file based on the environment.

        Parameters
        ----------
        config_file
            Configuration file name.

        Returns
        -------
        DotMap:
            Parsed configuration mapping.
        """

        with open(Path("/home/hadoop").joinpath(config_file), "r") as f:

            contents = yaml.safe_load(f)
            config = contents["global"]
            account = contents[self.accountId]["default"]
            region = contents[self.accountId][self.region]

            self.update_config(account, config)
            self.update_config(region, config)

            self.display = DotMap(config.copy())
            return DotMap(config)



[docs]    def update_config(self, update: dict, config: dict):
        """
        Helper method for dictionary update and insert.

        Parameters
        ----------
        update
            Dictionary to merge.
        config
            Dictionary to update.
        """

        for k, v in update.items():
            if k in config:
                config[k].update(v)
            else:
                config[k] = v



[docs]    def load_secrets(self, config: DotMap, display: DotMap):
        """
        Load secrets into configuration mapping.

        Parameters
        ----------
        config
            Configuration mapping.
        display
            Display mapping (sensitive values are obscured).
        """

        for k, v in config.items():
            if isinstance(v, dict):
                if set(v.keys()) == {"path", "key"}:
                    config[k] = self.vault.read(v.path, v.key)
                    display[k] = self.obscure(config[k])
                else:
                    self.load_secrets(v, display[k])



[docs]    def obscure(self, value: str, peek: int = 4) -> str:
        """
        Helper method for obscuring strings.

        Parameters
        ----------
        value
            String to obscure.
        peek: int, optional
            Number of characters to keep visible.
        """

        return "*" * (len(value) - peek) + value[-peek:]



[docs]    def get_spark_session(self) -> SparkSession:
        """
        Get spark session.
        """

        spark = (
            SparkSession.builder.master("local[*]")
            .config("spark.driver.extraClassPath", self.config.constants["SPARK_JARS"])
            .config(
                "spark.executor.extraClassPath", self.config.constants["SPARK_JARS"]
            )
            .config(conf=make_q2_sparkconf())
            .getOrCreate()
        )
        patch_q2_session(spark)
        return spark
